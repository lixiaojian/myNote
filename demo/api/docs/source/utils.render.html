var UTIL = UTIL || {};
UTIL.render = UTIL.render || {
	genItemTpl: function(item, tpl, reFunc, cxt){
		return tpl.replace(/\{[a-zA-z]*\}/ig, function(w){
			var key = w.substr(1, w.length - 2);
			var str = item[key] ? item[key] : '';
			if(reFunc)
				str = reFunc.call(cxt || this, item, key, str);
			return str;
		});
	}
};
