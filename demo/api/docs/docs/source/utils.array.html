var UTIL = UTIL || {};
UTIL.array = UTIL.array || {
	genDataByMedian: function(m, l, r){ //m: Median, l: left range, r: right range
		var arr = [];
		var i = 0;
		l = (+l);
		r = (+r);
		if(isNaN(l) && !l) l = 0;
		if(isNaN(r) && !r) r = l;
		i = l;
		while(i > 0){
			arr.push(m - i);
			i--;
		}
		arr.push(m);
		i = 1;
		r = r + 1;
		while(i < r){
			arr.push(m + i);
			i++;
		}
		return arr;
	}
};
